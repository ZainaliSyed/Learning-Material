    public static int updateEmployRecord(int employid, byte[] img)throws Exception{
        String query="Update EmployeeTable set image=? where employee_id=?";
        System.out.println(query);     
        PreparedStatement ps = null;
        try{
            ps = con.prepareStatement(query);    
            ps.setBytes(1,img);     
            ps.setInt(2,employid);
            int rows= ps.executeUpdate();
            con.commit();
            return rows;
        }
